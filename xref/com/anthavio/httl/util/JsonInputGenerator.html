77  	public AstNode parse(String className, Reader reader) throws IOException {
78  		ObjectMapper mapper = new ObjectMapper();
79  		JsonFactory factory = mapper.getFactory();
80  		JsonParser parser = factory.createJsonParser(reader);
81  		String xName = className + System.currentTimeMillis();
82  		stack.add(new AstNode(xName, null, false)); //artificial root element - must stay on the top of the Stack
83  		doField(className, parser);
84  		if (!xName.equals(stack.peek().getName())) {
85  			throw new IllegalStateException("Artificial element " + xName + " not found on the top of the stack");
86  		}
87  		AstNode rootEntry = stack.pop().getElements().get(0);
88  		if (!rootEntry.getName().equals(className)) {
89  			throw new IllegalStateException("Root element " + className + " does not match with " + rootEntry.getName());
90  		}
91  		return rootEntry;
92  	}

94  	private void doField(String name, JsonParser parser) throws IOException {
95  		logger.debug("Field begin " + name);
96  		JsonToken token = parser.nextToken();
97  		if (token == JsonToken.START_OBJECT) {
98  			doObject(name, parser);
99  		} else if (token == JsonToken.START_ARRAY) {
100 			doArray(name, parser);
101 		} else { //must be value then
102 			Class<?> type = doValue(parser);
103 			stack.peek().addField(name, type);
104 		}
105 		logger.debug("Field ended " + name);
106 	}

108 	private AstNode doObject(String name, JsonParser parser) throws IOException {
109 		logger.debug("Object begin " + name);
110 		AstNode object = AstNode.object(name);
111 		stack.push(object);
112 		while (parser.nextToken() != JsonToken.END_OBJECT) {
113 			JsonToken token = parser.getCurrentToken();
114 			if (token == JsonToken.FIELD_NAME) {
115 				String fieldName = parser.getCurrentName();
116 				doField(fieldName, parser);
117 			} else {
118 				throw new IllegalArgumentException("Unexpected token " + token);
119 			}
120 		}
121 		if (stack.pop() != object) {
122 			throw new IllegalStateException("Stack corrupted for " + object);
123 		}
124 		//System.out.println(object.getName() + " " + object.getTypeName());
125 		object = declare(object);
126 
127 		stack.peek().addField(object); //add to parent
128 		logger.debug("Object ended " + name + " " + object);
129 		return object;
130 	}

132 	private AstNode declare(AstNode object) {
133 		logger.debug("declare " + object);
134 		List<AstNode> objects;
135 		if (doGlobalTypes) {
136 			objects = globals;
137 		} else { //!doGlobalTypes
138 			AstNode parent = stack.peek();
139 			int s = stack.size();
140 			while (parent.isArray()) {
141 				parent = stack.get(--s); //cannot declare in array
142 			}
143 			objects = parent.locals;
144 		}
145 
146 		//if we already have object with same set of fields - use it instead of creating new
147 		for (AstNode existing : objects) {
148 			//System.out.println(object + " vs " + existing);
149 			if (existing.equalsFields(object)) {
150 				//System.out.println("existing " + object);
151 				return existing;
152 			}
153 		}
154 
155 		//object with same set of fields not found - create new
156 		String typeName = buildClassName(object.getName());
157 		//check for name collision
158 		for (AstNode existing : objects) {
159 			if (existing.getName().equals(typeName)) {
160 				typeName = stack.peek().getName() + typeName; //use enclosing object name as prefix
161 			}
162 		}
163 		object.setTypeName(typeName);
164 		objects.add(object);
165 
166 		return object;
167 	}

169 	private AstNode doArray(String name, JsonParser parser) throws IOException {
170 		logger.debug("Array begin " + name);
171 		AstNode array = AstNode.array(name);
172 		stack.push(array);
173 		int counter = 0;
174 		AstNode first = null;
175 		while (parser.nextToken() != JsonToken.END_ARRAY) {
176 			++counter;
177 			JsonToken token = parser.getCurrentToken();
178 			//We don't know type name for array element - create artificial
179 			//String eName = array.getName();
180 			AstNode element;
181 			if (token == JsonToken.START_OBJECT) {
182 				element = doObject(name, parser);
183 			} else if (token == JsonToken.START_ARRAY) {
184 				element = doArray(name, parser);
185 			} else { //must be value then
186 				Class<?> type = doValue(parser);
187 				if (type == Void.class) {
188 					continue; //skip JSON null elements
189 				}
190 				element = array.addField(name, type);
191 			}
192 
193 			if (first == null) {
194 				first = element;
195 				//System.out.println("array will be " + element.getTypeName());
196 				array.setTypeName(element.getTypeName());
197 			} else if (!first.equalsFields(element)) {
198 				if (first.isSimpleType() && element.isSimpleType()) {
199 					//if heterogenity is for example Integer vs String, we can still use to String...
200 					logger.debug("Heterogenous simple type array " + name + " of " + first.getTypeName() + " vs "
201 							+ element.getTypeName());
202 					array.setTypeName(String.class);
203 				} else {
204 					if ((first.isSimpleType() && !element.isSimpleType()) || (!first.isSimpleType() && element.isSimpleType())) {
205 						throw new IllegalStateException("Mixing simple and complex types in array is not allowed: "
206 								+ first.getTypeName() + " vs " + element.getTypeName());
207 					}
208 					logger.warn("Heterogenous complex type array " + name + " of " + first.getTypeName() + " vs "
209 							+ element.getTypeName());
210 					first.setTypeName(first.getTypeName() + "1");//distinguish class names
211 					element.setTypeName(element.getTypeName() + counter);
212 					array.setTypeName(Map.class); //Jackson can parse anything into Map
213 				}
214 			}
215 		}
216 
217 		if (stack.pop() != array) {
218 			throw new IllegalStateException("Stack corrupted for " + array);
219 		}
220 		stack.peek().addField(array); //add this array into parent object
221 		logger.debug("Array ended " + name + " " + array);
222 		return array;
223 	}

228 	private Class<?> doValue(JsonParser parser) throws IOException {
229 		Class<?> type;
230 		JsonToken token = parser.getCurrentToken();
231 		String value = parser.getValueAsString();
232 		switch (token) {
233 		case VALUE_STRING:
234 			type = String.class;
235 			break;
236 		case VALUE_NUMBER_INT:
237 			type = Integer.class;
238 			break;
239 		case VALUE_NUMBER_FLOAT:
240 			type = Float.class;
241 			break;
242 		case VALUE_TRUE:
243 			type = Boolean.class;
244 			break;
245 		case VALUE_FALSE:
246 			type = Boolean.class;
247 			break;
248 		case VALUE_NULL:
249 			type = Void.class;
250 			break;
251 		default:
252 			throw new IllegalArgumentException("Unexpected token " + token);
253 		}
254 		if (type == String.class && dateFormat != null) {
255 			try {
256 				dateFormat.parse(value);
257 				type = Date.class;
258 			} catch (ParseException px) {
259 				//not a date, keep it string
260 			}
261 		}
262 		logger.debug("Value " + value + " of " + type);
263 		return type;
264 	}
