92  		protected String buildFieldName(String name) {
93  			char c0 = name.charAt(0);
94  			if (Character.isDigit(c0)) { //OK in JSON, NOT in Java
95  				name = "f" + name;
96  			} else if (Character.isUpperCase(c0)) {
97  				if (name.length() == 1) {
98  					name = Character.toLowerCase(c0) + "";
99  				} else {
100 					name = Character.toLowerCase(c0) + name.substring(1);
101 				}
102 			}
103 			return name;
104 		}

121 			StringBuilder sb = new StringBuilder();
122 			AstNode f = field;
123 			while (f.isArray()) {
124 				sb.append("java.util.List<");
125 				f = f.getElements().get(0); //JSON null?
126 			}
127 
128 			sb.append(field.getTypeName());
129 
130 			f = field;

132 				f = f.getElements().get(0); //JSON null?

139 		public void write(AstNode root, Writer writer) {
140 
141 			IndentingWriter w = new IndentingWriter(writer);
142 			if (Cutils.isNotEmpty(targetPackage)) {
143 				w.println("package " + targetPackage + ";");
144 			}
145 			//pw.println("import java.util.List;");
146 			//pw.println("import java.util.Map;");
147 			//pw.println("import java.util.Date;");
148 			w.println("/**");
149 			w.println(" * Generated by Hatatitla at " + new SimpleDateFormat("yyyy.MM.dd HH:mm:ss").format(new Date()));
150 			w.println(" */");
151 			if (root.isArray()) {
152 
153 				if (root.getTypeName().equals("java.util.Map")) {
154 					//heretogenous array - elements has different types
155 					List<AstNode> elements = root.getElements();
156 					for (AstNode element : elements) {
157 						writeClass(element, Modifier.PUBLIC, w);
158 					}
159 				} else {
160 					//homogenous array - type is same so just first
161 					writeClass(root.getElements().get(0), Modifier.PUBLIC, w);
162 				}
163 			} else {
164 				writeClass(root, Modifier.PUBLIC, w);
165 			}
166 
167 			for (AstNode global : globals) {
168 				writeClass(global, 0, w);
169 			}

173 		private void writeClass(AstNode node, int modifiers, IndentingWriter w) {
174 			if (Modifier.isPublic(modifiers)) {
175 				w.print("public ");
176 			} else if (Modifier.isProtected(modifiers)) {
177 				w.print("protected ");
178 			} else if (Modifier.isProtected(modifiers)) {
179 				w.print("private ");
180 			}
181 			if (Modifier.isStatic(modifiers)) {
182 				w.print("static ");
183 			}
184 			if (Modifier.isFinal(modifiers)) {
185 				w.print("final ");
186 			}
187 			w.print("class " + node.getTypeName());
188 
189 			if (Cutils.isNotEmpty(baseClass)) {
190 				w.print(" extends " + baseClass);
191 			}
192 			if (interfaces.size() != 0) {
193 				w.print(" implements ");
194 				for (int i = 0; i < interfaces.size(); ++i) {
195 					String itf = interfaces.get(i);
196 					w.print(itf);
197 					if (i < interfaces.size() - 1) {
198 						w.print(", ");
199 					}
200 				}
201 			}
202 			w.print(" {");
203 			w.println();
204 			w.incLevel();
205 
206 			List<AstNode> fields = node.getElements();
207 			for (AstNode field : fields) {
208 				writeFieldDeclaration(w, field);
209 			}
210 
211 			for (AstNode field : fields) {
212 				writeFieldGetSet(w, field);
213 			}
214 
215 			//local types
216 			for (AstNode local : node.locals) {
217 				writeClass(local, Modifier.PUBLIC | Modifier.STATIC, w);
218 			}
219 
220 			w.decLevel();
221 			w.println("}");
222 		}

224 		private void writeFieldGetSet(IndentingWriter w, AstNode field) {
225 			String type = buildFieldType(field);
226 			String fieldName = buildFieldName(field.getName());
227 			String name;
228 			if (fieldName.length() == 1) {
229 				name = Character.toUpperCase(fieldName.charAt(0)) + "";
230 			} else {
231 				name = Character.toUpperCase(fieldName.charAt(0)) + fieldName.substring(1);
232 			}
233 			//getter
234 			w.print("public ");
235 			w.print(type);
236 			w.print(" get" + name + "() {");
237 			w.incLevel();
238 			w.println();
239 			w.println("return this." + fieldName + ";");
240 			w.decLevel();
241 			w.println("}");
242 			//setter
243 			w.print("public void ");
244 			w.print(" set" + name + "(" + type + " " + fieldName + ") {");
245 			w.incLevel();
246 			w.println();
247 			w.println("this." + fieldName + " = " + fieldName + ";");
248 			w.decLevel();
249 			w.println("}");
250 
251 		}

342 			if (Cutils.isEmpty(className)) {
343 				throw new IllegalArgumentException("rootName is empty");
344 			}
345 			int dotIdx = className.lastIndexOf('.');
346 			if (dotIdx != -1) {
347 				targetPackage = className.substring(0, dotIdx);
348 				className = className.substring(dotIdx + 1);
349 			}
350 			AstNode root = parse(className, reader);
351 			//System.out.println(root);
352 			StringWriter sw = new StringWriter();
353 			write(root, sw);
354 			return sw.toString();
355 		}

357 		public void compile(String className, String javaCode) throws IOException {
358 			String fileName;
359 			int dotIdx = className.lastIndexOf('.');
360 			if (dotIdx != -1) {
361 				fileName = className.substring(dotIdx + 1) + ".java";
362 			} else {
363 				fileName = className;
364 			}
365 			if (!fileName.endsWith(".java")) {
366 				fileName += ".java";
367 			}
368 			logger.debug("Compiling " + fileName);
369 			//javax.tool compiler api is cool as hell
370 			JavaCompiler compiler = ToolProvider.getSystemJavaCompiler();
371 			DiagnosticCollector<JavaFileObject> diagnosticListener = new DiagnosticCollector<JavaFileObject>();
372 			StandardJavaFileManager fileManager = compiler.getStandardFileManager(diagnosticListener, Locale.getDefault(),
373 					Charset.forName("utf-8"));
374 			JavaFileObject java = new StringJavaObject(fileName, javaCode);
375 			List<JavaFileObject> compilationUnits = Arrays.asList(java);
376 
377 			//otherwise classes are written to current directory
378 			String[] options = new String[] { "-d", "target/classes" };
379 			List<String> compileOptions = Arrays.asList(options);
380 
381 			CompilationTask task = compiler.getTask(null, fileManager, diagnosticListener, compileOptions, null,
382 					compilationUnits);
383 			Boolean result = task.call();
384 			/*
385 			List<Diagnostic<? extends JavaFileObject>> diagnostics = diagnosticListener.getDiagnostics();
386 			for (Diagnostic<? extends JavaFileObject> diagnosticItem : diagnostics) {
387 				System.out.format("Error in %s", diagnosticItem);
388 			}
389 			*/
390 			fileManager.close();
391 
392 			if (!result) {
393 				throw new IllegalArgumentException("Compilation failed " + diagnosticListener.getDiagnostics());
394 			}
395 
396 		}

398 		private AstNode parse(String className, Reader reader) throws IOException, JsonParseException {
399 			ObjectMapper mapper = new ObjectMapper();
400 			JsonFactory factory = mapper.getFactory();
401 			JsonParser parser = factory.createJsonParser(reader);
402 			String xName = className + System.currentTimeMillis();
403 			stack.add(new AstNode(xName, null, false)); //artificial root element - must stay on the top of the Stack
404 			doField(className, parser);
405 			if (!xName.equals(stack.peek().getName())) {
406 				throw new IllegalStateException("Artificial element " + xName + " not found on the top of the stack");
407 			}
408 			AstNode rootEntry = stack.pop().getElements().get(0);
409 			if (!rootEntry.getName().equals(className)) {
410 				throw new IllegalStateException("Root element " + className + " does not match with " + rootEntry.getName());
411 			}
412 			return rootEntry;
413 		}

415 		private void doField(String name, JsonParser parser) throws IOException {
416 			logger.debug("Field begin " + name);
417 			JsonToken token = parser.nextToken();
418 			if (token == JsonToken.START_OBJECT) {
419 				doObject(name, parser);
420 			} else if (token == JsonToken.START_ARRAY) {
421 				doArray(name, parser);
422 			} else { //must be value then
423 				Class<?> type = doValue(parser);
424 				stack.peek().addField(name, type);
425 			}
426 			logger.debug("Field ended " + name);
427 		}

429 		private AstNode doObject(String name, JsonParser parser) throws IOException {
430 			logger.debug("Object begin " + name);
431 			AstNode object = AstNode.object(name);
432 			stack.push(object);
433 			while (parser.nextToken() != JsonToken.END_OBJECT) {
434 				JsonToken token = parser.getCurrentToken();
435 				if (token == JsonToken.FIELD_NAME) {
436 					String fieldName = parser.getCurrentName();
437 					doField(fieldName, parser);
438 				} else {
439 					throw new IllegalArgumentException("Unexpected token " + token);
440 				}
441 			}
442 			if (stack.pop() != object) {
443 				throw new IllegalStateException("Stack corrupted for " + object);
444 			}
445 			//System.out.println(object.getName() + " " + object.getTypeName());
446 			object = declare(object);
447 
448 			stack.peek().addField(object); //add to parent
449 			logger.debug("Object ended " + name + " " + object);
450 			return object;
451 		}

453 		private AstNode declare(AstNode object) {
454 			logger.debug("declare " + object);
455 			List<AstNode> objects;
456 			if (doGlobalTypes) {
457 				objects = globals;
458 			} else { //!doGlobalTypes
459 				AstNode parent = stack.peek();
460 				int s = stack.size();
461 				while (parent.isArray()) {
462 					parent = stack.get(--s); //cannot declare in array
463 				}
464 				objects = parent.locals;
465 			}
466 
467 			//if we already have object with same set of fields - use it instead of creating new
468 			for (AstNode existing : objects) {
469 				//System.out.println(object + " vs " + existing);
470 				if (existing.equalsFields(object)) {
471 					//System.out.println("existing " + object);
472 					return existing;
473 				}
474 			}
475 
476 			//object with same set of fields not found - create new
477 			String typeName = buildClassName(object.getName());
478 			//check for name collision
479 			for (AstNode existing : objects) {
480 				if (existing.getName().equals(typeName)) {
481 					typeName = stack.peek().getName() + typeName; //use enclosing object name as prefix
482 				}
483 			}
484 			object.setTypeName(typeName);
485 			objects.add(object);
486 
487 			return object;
488 		}

490 		private AstNode doArray(String name, JsonParser parser) throws IOException {
491 			logger.debug("Array begin " + name);
492 			AstNode array = AstNode.array(name);
493 			stack.push(array);
494 			int counter = 0;
495 			AstNode first = null;
496 			while (parser.nextToken() != JsonToken.END_ARRAY) {
497 				++counter;
498 				JsonToken token = parser.getCurrentToken();
499 				//We don't know type name for array element - create artificial
500 				//String eName = array.getName();
501 				AstNode element;
502 				if (token == JsonToken.START_OBJECT) {
503 					element = doObject(name, parser);
504 				} else if (token == JsonToken.START_ARRAY) {
505 					element = doArray(name, parser);
506 				} else { //must be value then
507 					Class<?> type = doValue(parser);
508 					if (type == Void.class) {
509 						continue; //skip JSON null elements
510 					}
511 					element = array.addField(name, type);
512 				}
513 
514 				if (first == null) {
515 					first = element;
516 					//System.out.println("array will be " + element.getTypeName());
517 					array.setTypeName(element.getTypeName());
518 				} else if (!first.equalsFields(element)) {
519 					if (first.isSimpleType() && element.isSimpleType()) {
520 						//if heterogenity is for example Integer vs String, we can still use to String...
521 						logger.debug("Heterogenous simple type array " + name + " of " + first.getTypeName() + " vs "
522 								+ element.getTypeName());
523 						array.setTypeName(String.class);
524 					} else {
525 						if ((first.isSimpleType() && !element.isSimpleType()) || (!first.isSimpleType() && element.isSimpleType())) {
526 							throw new IllegalStateException("Mixing simple and complex types in array is not allowed: "
527 									+ first.getTypeName() + " vs " + element.getTypeName());
528 						}
529 						logger.warn("Heterogenous complex type array " + name + " of " + first.getTypeName() + " vs "
530 								+ element.getTypeName());
531 						first.setTypeName(first.getTypeName() + "1");//distinguish class names
532 						element.setTypeName(element.getTypeName() + counter);
533 						array.setTypeName(Map.class); //Jackson can parse anything into Map
534 					}
535 				}
536 			}
537 
538 			if (stack.pop() != array) {
539 				throw new IllegalStateException("Stack corrupted for " + array);
540 			}
541 			stack.peek().addField(array); //add this array into parent object
542 			logger.debug("Array ended " + name + " " + array);
543 			return array;
544 		}

546 		/**
547 		 * XXX maybe provide some value parser / type resolver for pluggable types instead of this hardcoded stuff
548 		 */
549 		private Class<?> doValue(JsonParser parser) throws IOException {
550 			Class<?> type;
551 			JsonToken token = parser.getCurrentToken();
552 			String value = parser.getValueAsString();
553 			switch (token) {
554 			case VALUE_STRING:
555 				type = String.class;
556 				break;
557 			case VALUE_NUMBER_INT:
558 				type = Integer.class;
559 				break;
560 			case VALUE_NUMBER_FLOAT:
561 				type = Float.class;
562 				break;
563 			case VALUE_TRUE:
564 				type = Boolean.class;
565 				break;
566 			case VALUE_FALSE:
567 				type = Boolean.class;
568 				break;
569 			case VALUE_NULL:
570 				type = Void.class;
571 				break;
572 			default:
573 				throw new IllegalArgumentException("Unexpected token " + token);
574 			}
575 			if (type == String.class && dateFormat != null) {
576 				try {
577 					dateFormat.parse(value);
578 					type = Date.class;
579 				} catch (ParseException px) {
580 					//not a date, keep it string
581 				}
582 			}
583 			logger.debug("Value " + value + " of " + type);
584 			return type;
585 		}
