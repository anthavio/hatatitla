134 
135 		IndentingWriter w = new IndentingWriter(writer);
136 		if (Cutils.isNotEmpty(targetPackage)) {
137 			w.println("package " + targetPackage + ";");
138 		}
139 		//pw.println("import java.util.List;");
140 		//pw.println("import java.util.Map;");
141 		//pw.println("import java.util.Date;");
142 		w.println("/**");
143 		w.println(" * Generated by Hatatitla at " + new SimpleDateFormat("yyyy.MM.dd HH:mm:ss").format(new Date()));
144 		w.println(" */");
145 		if (root.isArray()) {
146 
147 			if (root.getTypeName().equals("java.util.Map")) {
148 				//heretogenous array - elements has different types
149 				List<AstNode> elements = root.getElements();
150 				for (AstNode element : elements) {
151 					writeClass(element, Modifier.PUBLIC, w);
152 				}
153 			} else {
154 				//homogenous array - type is same so just first
155 				writeClass(root.getElements().get(0), Modifier.PUBLIC, w);
156 			}
157 		} else {
158 			writeClass(root, Modifier.PUBLIC, w);
159 		}
160 
161 		for (AstNode global : globals) {
162 			writeClass(global, 0, w);
163 		}

167 	private void writeClass(AstNode node, int modifiers, IndentingWriter w) {
168 		if (Modifier.isPublic(modifiers)) {
169 			w.print("public ");
170 		} else if (Modifier.isProtected(modifiers)) {
171 			w.print("protected ");
172 		} else if (Modifier.isProtected(modifiers)) {
173 			w.print("private ");
174 		}
175 		if (Modifier.isStatic(modifiers)) {
176 			w.print("static ");
177 		}
178 		if (Modifier.isFinal(modifiers)) {
179 			w.print("final ");
180 		}
181 		w.print("class " + node.getTypeName());
182 
183 		if (Cutils.isNotEmpty(baseClass)) {
184 			w.print(" extends " + baseClass);
185 		}
186 		if (interfaces != null && interfaces.size() != 0) {
187 			w.print(" implements ");
188 			for (int i = 0; i < interfaces.size(); ++i) {
189 				String itf = interfaces.get(i);
190 				w.print(itf);
191 				if (i < interfaces.size() - 1) {
192 					w.print(", ");
193 				}
194 			}
195 		}
196 		w.print(" {");
197 		w.println();
198 		w.incLevel();
199 
200 		List<AstNode> fields = node.getElements();
201 		for (AstNode field : fields) {
202 			writeFieldDeclaration(w, field);
203 		}
204 
205 		for (AstNode field : fields) {
206 			writeFieldGetSet(w, field);
207 		}
208 
209 		//local types
210 		for (AstNode local : node.locals) {
211 			writeClass(local, Modifier.PUBLIC | Modifier.STATIC, w);
212 		}
213 
214 		w.decLevel();
215 		w.println("}");
216 	}

218 	private void writeFieldGetSet(IndentingWriter w, AstNode field) {
219 		String type = buildFieldType(field);
220 		String fieldName = buildFieldName(field.getName());
221 		String name;
222 		if (fieldName.length() == 1) {
223 			name = Character.toUpperCase(fieldName.charAt(0)) + "";
224 		} else {
225 			name = Character.toUpperCase(fieldName.charAt(0)) + fieldName.substring(1);
226 		}
227 		//getter
228 		w.print("public ");
229 		w.print(type);
230 		w.print(" get" + name + "() {");
231 		w.incLevel();
232 		w.println();
233 		w.println("return this." + fieldName + ";");
234 		w.decLevel();
235 		w.println("}");
236 		//setter
237 		w.print("public void ");
238 		w.print(" set" + name + "(" + type + " " + fieldName + ") {");
239 		w.incLevel();
240 		w.println();
241 		w.println("this." + fieldName + " = " + fieldName + ";");
242 		w.decLevel();
243 		w.println("}");
