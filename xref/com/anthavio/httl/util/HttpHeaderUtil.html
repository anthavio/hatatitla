21  	public static CacheEntry<CachedResponse> buildCacheEntry(SenderRequest request, SenderResponse response) {
22  
23  		Multival headers = response.getHeaders();
24  
25  		long softTtl = 0; //seconds
26  		long maxAge = 0; //seconds
27  		boolean hasCacheControl = false;
28  
29  		String headerValue = headers.getFirst("Cache-Control");
30  		if (headerValue != null) {
31  			hasCacheControl = true;
32  			String[] tokens = headerValue.split(",");
33  			for (int i = 0; i < tokens.length; i++) {
34  				String token = tokens[i].trim();
35  				if (token.equals("no-store") || token.equals("no-cache")) {
36  					//always check server for new version (with If-None-Match (ETag) or If-Modified-Since(Last-Modified/Date))
37  					//if ETag or Last-Modified is not present, we will not cache this reponse at all
38  					maxAge = 0;
39  					break;
40  				} else if (token.equals("must-revalidate") || token.equals("proxy-revalidate")) {
41  					//cache response until expire (max-age, Expires)
42  					//then check server for new version (with If-None-Match (ETag) or If-Modified-Since(Last-Modified/Date))
43  					//when max-age=0 or Expires=-1 then this is same as no-store/no-cache
44  					maxAge = 0;
45  				} else if (token.startsWith("max-age=")) {
46  					try {
47  						maxAge = Long.parseLong(token.substring(8));//seconds
48  						break;
49  					} catch (Exception e) {
50  						//ignore
51  					}
52  				}
53  			}
54  		}
55  
56  		long serverDate = 0;
57  		headerValue = headers.getFirst("Date");
58  		if (headerValue != null) {
59  			serverDate = parseDateAsEpoch(headerValue);
60  		}
61  
62  		long serverExpiry = 0;
63  		headerValue = headers.getFirst("Expires");
64  		if (headerValue != null) {
65  			serverExpiry = parseDateAsEpoch(headerValue);
66  		}
67  
68  		long lastModified = 0;
69  		headerValue = headers.getFirst("Last-Modified");
70  		if (headerValue != null) {
71  			lastModified = parseDateAsEpoch(headerValue);
72  		}
73  		Date modified = lastModified > 0 ? new Date(lastModified) : null;
74  
75  		String etag = headers.getFirst("ETag");
76  
77  		// Cache-Control takes precedence over an Expires header, 
78  		// even if both exist and Expires is more restrictive.
79  		if (hasCacheControl) {
80  			softTtl = maxAge;
81  		} else if (serverDate > 0 && serverExpiry >= serverDate) {
82  			// Default semantic for Expire header in HTTP specification is softExpire.
83  			softTtl = serverExpiry - serverDate;
84  		}
85  
86  		//if already expired and we don't have anything to check new version - don't cache at all
87  		if (softTtl <= 0 && Cutils.isEmpty(etag) && lastModified <= 0) {
88  			return null;
89  		}
90  		long hardTtl = softTtl > 0 ? softTtl : 10; //XXX default hardTtl is 10 seconds - should be parametrized
91  
92  		CachedResponse cachedResponse = new CachedResponse(request, response);
93  		return new CacheEntry<CachedResponse>(cachedResponse, hardTtl, softTtl, etag, modified);
94  
95  	}

232 	public static String readAsString(SenderResponse response) throws IOException {
233 		if (response instanceof CachedResponse) {
234 			return ((CachedResponse) response).getAsString();
235 		}
236 		if (response.getStream() == null) {
237 			return null; //304 NOT MODIFIED
238 		}
239 
240 		int blength = getBufferLength(response);
241 		char[] buffer = new char[blength];
242 		Reader input = new InputStreamReader(response.getStream(), response.getCharset());
243 		StringWriter output = new StringWriter();
244 		int len = -1;
245 		try {
246 			while ((len = input.read(buffer)) != -1) {
247 				output.write(buffer, 0, len);
248 			}
249 		} finally {
250 			input.close();
251 		}
252 		return output.toString();
253 	}
