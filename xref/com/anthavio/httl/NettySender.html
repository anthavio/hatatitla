90  	public ChannelPipeline retrieve(HttpRequest request) throws Exception {
91  		URI uri = new URI(request.getUri());
92  		int port = uri.getPort() == -1 ? 80 : uri.getPort();
93  		ChannelFuture connectFuture = client
94  				.connect(new InetSocketAddress(request.getHeader(HttpHeaders.Names.HOST), port));
95  		//connectFuture.addListener(new ConnectOk(request));
96  		connectFuture.awaitUninterruptibly();
97  		if (!connectFuture.isSuccess()) {
98  			connectFuture.getCause().printStackTrace();
99  			client.releaseExternalResources();
100 			throw new RuntimeException(connectFuture.getCause());
101 		}
102 		channels.add(connectFuture.getChannel());
103 		return connectFuture.getChannel().getPipeline();
104 	}

106 	private static class HttpClientPipelineFactory implements ChannelPipelineFactory {
107 
108 		private final boolean ssl;
109 
110 		public HttpClientPipelineFactory(boolean ssl) {
111 			this.ssl = ssl;
112 		}
113 
114 		@Override
115 		public ChannelPipeline getPipeline() throws Exception {
116 			ChannelPipeline pipeline = Channels.pipeline();
117 			Timer timer = new HashedWheelTimer();
118 			pipeline.addLast("timeout", new ReadTimeoutHandler(timer, 30));
119 			pipeline.addLast("codec", new HttpClientCodec());
120 
121 			if (ssl) {
122 				SSLContext sslContext = SSLContext.getInstance("TLS");
123 				//sslContext.init(keyManagers, trustManagers, null);
124 				SSLEngine engine = sslContext.createSSLEngine();
125 
126 				engine.setUseClientMode(true);
127 
128 				pipeline.addLast("ssl", new SslHandler(engine));
129 			}
130 
131 			//handle HttpChunks
132 			pipeline.addLast("aggregator", new HttpChunkAggregator(1048576));
133 			//automatic content decompression
134 			//pipeline.addLast("inflater", new HttpContentDecompressor());
135 			pipeline.addLast("handler", new HttpResponseHandler());
136 			return pipeline;
137 		}

144 		@Override

150 				System.out.println("VERSION: " + response.getProtocolVersion());
151 				System.out.println();
152 
153 				if (!response.getHeaderNames().isEmpty()) {
154 					for (String name : response.getHeaderNames()) {
155 						for (String value : response.getHeaders(name)) {
156 							System.out.println("HEADER: " + name + " = " + value);
157 						}
158 					}
159 					System.out.println();
160 				}
161 
162 				if (response.isChunked()) {
163 					readingChunks = true;
164 					System.out.println("CHUNKED CONTENT {");
165 				} else {
166 					ChannelBuffer content = response.getContent();
167 					if (content.readable()) {
168 						System.out.println("CONTENT {");
169 						System.out.println(content.toString(CharsetUtil.UTF_8));
170 						System.out.println("} END OF CONTENT");
171 					}
172 				}
173 			} else {
174 				HttpChunk chunk = (HttpChunk) e.getMessage();
175 				if (chunk.isLast()) {
176 					readingChunks = false;
177 					System.out.println("} END OF CHUNKED CONTENT");
178 				} else {
179 					System.out.print(chunk.getContent().toString(CharsetUtil.UTF_8));
180 					System.out.flush();
181 				}
182 			}
183 		}
