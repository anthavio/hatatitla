63  	public HttpClient buildHttpClient() {
64  		HttpParams httpParams = new BasicHttpParams();
65  		buildClientParams(httpParams, getHostUrl());
66  		buildConnectionParams(httpParams);
67  		buildProtocolParams(httpParams);
68  		ClientConnectionManager connectionManager = buildConnectionManager();
69  
70  		DefaultHttpClient httpClient = new DefaultHttpClient(connectionManager, httpParams);
71  
72  		if (getAuthentication() != null) {
73  			Authentication authentication = getAuthentication();
74  			UsernamePasswordCredentials credentials = new UsernamePasswordCredentials(authentication.getUsername(),
75  					authentication.getPassword());
76  			CredentialsProvider provider = httpClient.getCredentialsProvider();
77  			AuthScope scope = new AuthScope(getHostUrl().getHost(), getHostUrl().getPort(), authentication.getRealm()/*, authentication
78  																																																								.getScheme().toString()*/);
79  			provider.setCredentials(scope, credentials);
80  
81  			if (authentication.getPreemptive()) {
82  				AuthScheme scheme;
83  				if (authentication.getScheme() == Authentication.Scheme.BASIC) {
84  					BasicScheme basic = new BasicScheme();
85  					scheme = basic;
86  				} else {
87  					//http://stackoverflow.com/questions/2954434/apache-httpclient-digest-authentication
88  					DigestScheme digest = new DigestScheme();
89  					digest.overrideParamter("realm", authentication.getRealm());
90  					digest.overrideParamter("nonce", authentication.getNonce());
91  					scheme = digest;
92  				}
93  				HttpHost host = new HttpHost(getHostUrl().getHost(), getHostUrl().getPort(), getHostUrl().getProtocol());
94  				AuthCache authCache = new BasicAuthCache();
95  				authCache.put(host, scheme);
96  
97  				this.authContext = new BasicHttpContext();
98  				this.authContext.setAttribute(ClientContext.AUTH_CACHE, authCache);
99  				//authContext.setAttribute(ClientContext.CREDS_PROVIDER, provider);
100 			}
101 
102 		}
103 
104 		return httpClient;
105 	}

133 	protected PoolingClientConnectionManager buildConnectionManager() {
134 		SchemeRegistry schemeRegistry = new SchemeRegistry();
135 		schemeRegistry.register(new Scheme("http", 80, PlainSocketFactory.getSocketFactory()));
136 
137 		//set strict hostname verifier
138 		SSLContext sslcontext;
139 		try {
140 			sslcontext = SSLContext.getInstance("TLS");
141 			sslcontext.init(null, null, null);
142 		} catch (NoSuchAlgorithmException nsax) {
143 			throw new IllegalArgumentException(nsax);
144 		} catch (KeyManagementException kmx) {
145 			throw new IllegalArgumentException(kmx);
146 		}
147 		SSLSocketFactory sslSocketFactory = new SSLSocketFactory(sslcontext, SSLSocketFactory.STRICT_HOSTNAME_VERIFIER);
148 		schemeRegistry.register(new Scheme("https", 443, sslSocketFactory));
149 
150 		//we access only one host
151 		PoolingClientConnectionManager connectionManager = new PoolingClientConnectionManager(schemeRegistry,
152 				this.poolReleaseTimeoutMillis, TimeUnit.MILLISECONDS);
153 		connectionManager.setMaxTotal(getPoolMaximumSize());
154 		connectionManager.setDefaultMaxPerRoute(getPoolMaximumSize());
155 		return connectionManager;
