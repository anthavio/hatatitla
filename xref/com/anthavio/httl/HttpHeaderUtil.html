19  	public static CacheEntry<CachedResponse> buildCacheEntry(SenderRequest request, SenderResponse response) {
20  
21  		Multival headers = response.getHeaders();
22  
23  		long softTtl = 0; //seconds
24  		long maxAge = 0; //seconds
25  		boolean hasCacheControl = false;
26  
27  		String headerValue = headers.getFirst("Cache-Control");
28  		if (headerValue != null) {
29  			hasCacheControl = true;
30  			String[] tokens = headerValue.split(",");
31  			for (int i = 0; i < tokens.length; i++) {
32  				String token = tokens[i].trim();
33  				if (token.equals("no-store") || token.equals("no-cache")) {
34  					//always check server for new version (with If-None-Match (ETag) or If-Modified-Since(Last-Modified/Date))
35  					//if ETag or Last-Modified is not present, we will not cache this reponse at all
36  					maxAge = 0;
37  					break;
38  				} else if (token.equals("must-revalidate") || token.equals("proxy-revalidate")) {
39  					//cache response until expire (max-age, Expires)
40  					//then check server for new version (with If-None-Match (ETag) or If-Modified-Since(Last-Modified/Date))
41  					//when max-age=0 or Expires=-1 then this is same as no-store/no-cache
42  					maxAge = 0;
43  				} else if (token.startsWith("max-age=")) {
44  					try {
45  						maxAge = Long.parseLong(token.substring(8));//seconds
46  						break;
47  					} catch (Exception e) {
48  						//ignore
49  					}
50  				}
51  			}
52  		}
53  
54  		long serverDate = 0;
55  		headerValue = headers.getFirst("Date");
56  		if (headerValue != null) {
57  			serverDate = parseDateAsEpoch(headerValue);
58  		}
59  
60  		long serverExpiry = 0;
61  		headerValue = headers.getFirst("Expires");
62  		if (headerValue != null) {
63  			serverExpiry = parseDateAsEpoch(headerValue);
64  		}
65  
66  		long lastModified = 0;
67  		headerValue = headers.getFirst("Last-Modified");
68  		if (headerValue != null) {
69  			lastModified = parseDateAsEpoch(headerValue);
70  		}
71  		Date modified = lastModified > 0 ? new Date(lastModified) : null;
72  
73  		String etag = headers.getFirst("ETag");
74  
75  		// Cache-Control takes precedence over an Expires header, 
76  		// even if both exist and Expires is more restrictive.
77  		if (hasCacheControl) {
78  			softTtl = maxAge;
79  		} else if (serverDate > 0 && serverExpiry >= serverDate) {
80  			// Default semantic for Expire header in HTTP specification is softExpire.
81  			softTtl = serverExpiry - serverDate;
82  		}
83  
84  		//if already expired and we don't have anything to check new version - don't cache at all
85  		if (softTtl <= 0 && Cutils.isEmpty(etag) && lastModified <= 0) {
86  			return null;
87  		}
88  		long hardTtl = softTtl > 0 ? softTtl : 10; //XXX default hardTtl is 10 seconds - should be parametrized
89  
90  		CachedResponse cachedResponse = new CachedResponse(request, response);
91  		return new CacheEntry<CachedResponse>(cachedResponse, hardTtl, softTtl, etag, modified);
92  
93  	}

230 	public static String readAsString(SenderResponse response) throws IOException {
231 		if (response instanceof CachedResponse) {
232 			return ((CachedResponse) response).getAsString();
233 		}
234 		if (response.getStream() == null) {
235 			return null; //304 NOT MODIFIED
236 		}
237 
238 		int blength = getBufferLength(response);
239 		char[] buffer = new char[blength];
240 		Reader input = new InputStreamReader(response.getStream(), response.getCharset());
241 		StringWriter output = new StringWriter();
242 		int len = -1;
243 		try {
244 			while ((len = input.read(buffer)) != -1) {
245 				output.write(buffer, 0, len);
246 			}
247 		} finally {
248 			input.close();
249 		}
250 		return output.toString();
251 	}
