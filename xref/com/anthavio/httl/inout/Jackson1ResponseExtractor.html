39  	}

47  		Object object = null;
48  		try {

50  			return (T) object;
51  		} catch (ClassCastException ccx) {
52  			String message = "Cannot cast: " + object.getClass().getName() + " into: " + resultType.getName() + " value: "

55  		}
56  	}
