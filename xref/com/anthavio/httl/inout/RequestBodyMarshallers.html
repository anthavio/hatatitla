11  /**
12   * Storage for HttpSender's RequestBodyMarshallers. 
13   * JAXB marshaller is allways installed fro mime types text/xml and application/xml
14   * When Jackson is found in classpath then JSON marshalling is initialized as well for mime type application/json
15   * 
16   * If user is unhappy with default marshallers, he can create and configure his own RequestJaxbMarshaller and set it via
17   * httpSender.setMarshaller(customJaxbMarshaller, "application/xml");
18   * httpSender.setMarshaller(customJaxbMarshaller, "text/xml");
19   * 
20   * @author martin.vanek
21   *
22   */
23  public class RequestBodyMarshallers {
24  
25  	private final Logger logger = LoggerFactory.getLogger(getClass());
26  
27  	private Map<String, RequestBodyMarshaller> marshallers = new HashMap<String, RequestBodyMarshaller>();
28  
29  	/**
30  	 * Initiate Built in Marshallers
31  	 */
32  	public RequestBodyMarshallers() {
33  
34  		JaxbRequestMarshaller jaxbMarshaller = new JaxbRequestMarshaller();
35  		marshallers.put("text/xml", jaxbMarshaller);
36  		marshallers.put("application/xml", jaxbMarshaller);
37  		logger.debug("Adding RequestJaxbMarshaller for XML requests");
38  
39  		//Jackson support is optional
40  		try {
41  			Class.forName("com.fasterxml.jackson.databind.ObjectMapper");
42  			marshallers.put("application/json", new Jackson2RequestMarshaller());
43  			logger.debug("Adding RequestJackson2Marshaller for JSON requests");
44  		} catch (ClassNotFoundException cnf) {
45  			try {
46  				Class.forName("org.codehaus.jackson.map.ObjectMapper");
47  				marshallers.put("application/json", new Jackson1RequestMarshaller());
48  				logger.debug("Adding RequestJackson1Marshaller for JSON requests");
49  			} catch (ClassNotFoundException cnf2) {
50  				logger.debug("Jackson classes not found. Built in JSON requests support is off");
51  			}
52  		}
53  	}
54  
55  	public RequestBodyMarshaller getMarshaller(String mimeType) {
56  		return marshallers.get(mimeType);
57  	}
58  
59  	/**
60  	 * Register request body marshaller with provided mimeType
61  	 */
62  	public void setMarshaller(RequestBodyMarshaller marshaller, String mimeType) {
63  		if (Cutils.isBlank(mimeType)) {
64  			throw new IllegalArgumentException("mime type is blank");
65  		}
66  		if (marshaller == null) {
67  			throw new IllegalArgumentException("marshaller is null");
68  		}
69  		this.marshallers.put(mimeType, marshaller);
70  	}
71  
72  	/**
73  	 * Shortcut to marshall request body
74  	
75  	public String marshall(Object body, String mimeType) throws IOException {
76  		RequestBodyMarshaller marshaller = getMarshaller(mimeType);
77  		if (marshaller == null) {
78  			throw new IllegalArgumentException("Marshaller found for Content-Type '" + mimeType + "'");
79  		}
80  		return marshaller.marshall(body);
81  	}
82  	 */
83  
84  	@Override
85  	public String toString() {
86  		return "RequestMarshallerFactory [marshallers=" + marshallers + "]";
