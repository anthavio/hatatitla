11  /**
12   * Storage for HttpSender's RequestBodyMarshallers. 
13   * JAXB marshaller is allways installed for media types text/xml and application/xml
14   * When Jackson is found in classpath then JSON marshalling is initialized as well for media type application/json
15   * 
16   * If user is unhappy with default marshallers, he can create and configure his own RequestJaxbMarshaller and set it via
17   * httpSender.setMarshaller(customMarshaller, "application/xml");
18   * httpSender.setMarshaller(customMarshaller, "text/xml");
19   * 
20   * @author martin.vanek
21   *
22   */
23  public class RequestBodyMarshallers {
24  
25  	private final Logger logger = LoggerFactory.getLogger(getClass());
26  
27  	private Map<String, RequestBodyMarshaller> marshallers = new HashMap<String, RequestBodyMarshaller>();
28  
29  	/**
30  	 * Initiate Built in Marshallers
31  	 */
32  	public RequestBodyMarshallers() {
33  
34  		RequestBodyMarshaller marshaller = null;
35  
36  		//First try SimpleXml
37  		try {
38  			Class.forName("org.simpleframework.xml.core.Persister");
39  			marshaller = new SimpleXmlRequestMarshaller();
40  			setMarshaller(marshaller, "text/xml");
41  			setMarshaller(marshaller, "application/xml");
42  		} catch (ClassNotFoundException cnfx) {
43  			//nothing
44  		}
45  
46  		//Then try JAXB
47  		if (marshaller == null) {
48  			try {
49  				Class.forName("javax.xml.bind.JAXBContext"); //JAXB is not avaliable on Android
50  				marshaller = new JaxbRequestMarshaller();
51  				setMarshaller(marshaller, "text/xml");
52  				setMarshaller(marshaller, "application/xml");
53  			} catch (ClassNotFoundException cnfx) {
54  				//nothing
55  			}
56  		}
57  
58  		if (marshaller == null) {
59  			logger.debug("No XML binding library found. XML body requests support is disabled");
60  		}
61  
62  		marshaller = null;
63  		//JSON support
64  
65  		//First try Jackson 2
66  		try {
67  			Class.forName("com.fasterxml.jackson.databind.ObjectMapper");
68  			marshaller = new Jackson2RequestMarshaller();
69  			setMarshaller(marshaller, "application/json");
70  		} catch (ClassNotFoundException cnfx) {
71  			//nothing
72  		}
73  
74  		//Then try Jackson 1
75  		if (marshaller == null) {
76  			try {
77  				Class.forName("org.codehaus.jackson.map.ObjectMapper");
78  				marshaller = new Jackson1RequestMarshaller();
79  				setMarshaller(marshaller, "application/json");
80  			} catch (ClassNotFoundException cnfx) {
81  				//nothing
82  			}
83  		}
84  
85  		//Then try Gson
86  		if (marshaller == null) {
87  			try {
88  				Class.forName("com.google.gson.Gson");
89  				marshaller = new GsonRequestMarshaller();
90  				setMarshaller(marshaller, "application/json");
91  			} catch (ClassNotFoundException cnfx) {
92  				//nothing
93  			}
94  		}
95  
96  		if (marshaller == null) {
97  			logger.debug("No JSON binding library found. JSON body requests support is disabled");
98  		}
99  	}

105 	/**
106 	 * Register request body marshaller with provided mediaType
107 	 */
108 	public void setMarshaller(RequestBodyMarshaller marshaller, String mediaType) {
109 		if (Cutils.isBlank(mediaType)) {
110 			throw new IllegalArgumentException("media type is blank");
111 		}
112 		if (marshaller == null) {
113 			throw new IllegalArgumentException("marshaller is null");
114 		}
115 		logger.debug("Adding " + marshaller.getClass().getName() + " for " + mediaType);
116 		this.marshallers.put(mediaType, marshaller);
117 	}

119 	/**
120 	 * Shortcut to marshall request body
121 	
122 	public String marshall(Object body, String mediaType) throws IOException {
123 		RequestBodyMarshaller marshaller = getMarshaller(mediaType);
124 		if (marshaller == null) {
125 			throw new IllegalArgumentException("Marshaller found for Content-Type '" + mediaType + "'");
126 		}
127 		return marshaller.marshall(body);
128 	}
129 	 */
