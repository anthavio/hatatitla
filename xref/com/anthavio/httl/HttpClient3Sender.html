81  	@Override
82  	public HttpClient3Response doExecute(SenderRequest request, String path, String query) throws IOException {
83  
84  		HttpMethodBase httpMethod;
85  		switch (request.getMethod()) {
86  		case GET:
87  			httpMethod = new GetMethod(path);
88  			break;
89  		case DELETE:
90  			httpMethod = new DeleteMethod(path);
91  			break;
92  		case POST:
93  			PostMethod httpPost = new PostMethod(path);
94  			RequestEntity entityPost = buildEntity(request, query);
95  			httpPost.setRequestEntity(entityPost);
96  			httpMethod = httpPost;
97  			break;
98  		case PUT:
99  			PutMethod httpPut = new PutMethod(path);
100 			RequestEntity entityPut = buildEntity(request, query);
101 			httpPut.setRequestEntity(entityPut);
102 			httpMethod = httpPut;
103 			break;
104 		default:
105 			throw new IllegalArgumentException("Unsupported method " + request.getMethod());
106 		}
107 
108 		Multival headers = request.getHeaders();
109 		if (headers != null && headers.size() != 0) {
110 			for (String name : headers) {
111 				List<String> values = headers.get(name);
112 				for (String value : values) {
113 					httpMethod.addRequestHeader(name, value);
114 				}
115 			}
116 		}
117 
118 		if (request.getReadTimeoutMillis() != null) {
119 			httpMethod.getParams().setParameter(HttpMethodParams.SO_TIMEOUT, request.getReadTimeoutMillis());
120 		}
121 		//cannot be set globally in configuration
122 		httpMethod.setFollowRedirects(config.getFollowRedirects());
123 
124 		if (config.getGzipRequest()) {
125 			httpMethod.addRequestHeader("Accept-Encoding", "gzip, deflate");
126 		}
127 
128 		if (request.hasBody()) {
129 			String contentType = request.getFirstHeader("Content-Type");
130 			if (contentType == null) {
131 				throw new IllegalArgumentException("Request with body must have Content-Type header specified");
132 			}
133 			//add charset into Content-Type header if missing
134 			int idxCharset = contentType.indexOf("charset=");
135 			if (idxCharset == -1) {
136 				contentType = contentType + "; charset=" + config.getCharset();
137 				httpMethod.setRequestHeader("Content-Type", contentType);
138 			}
139 		}
140 
141 		if (request.getFirstHeader("Accept") == null && config.getDefaultAccept() != null) {
142 			httpMethod.addRequestHeader("Accept", config.getDefaultAccept());
143 		}
144 
145 		if (request.getFirstHeader("Accept-Charset") == null) {
146 			httpMethod.addRequestHeader("Accept-Charset", config.getEncoding());
147 		}
148 
149 		int statusCode = call(httpMethod);
150 
151 		Header[] responseHeaders = httpMethod.getResponseHeaders();
152 
153 		Multival outHeaders = new Multival();
154 		for (Header header : responseHeaders) {
155 			outHeaders.add(header.getName(), header.getValue());
156 		}
157 		StatusLine statusLine = httpMethod.getStatusLine();
158 
159 		InputStream responseStream = httpMethod.getResponseBodyAsStream();
160 
161 		HttpClient3Response response = new HttpClient3Response(statusCode, statusLine.getReasonPhrase(), outHeaders,
162 				responseStream, httpMethod);
163 		return response;
164 	}

166 	private RequestEntity buildEntity(SenderRequest request, String query) throws IOException {
167 		String contentType = request.getFirstHeader("Content-Type");
168 		Object[] type = HttpHeaderUtil.splitContentType(contentType, config.getCharset());
169 		String mimeType = (String) type[0];
170 		Charset charset = (Charset) type[1];
171 
172 		RequestEntity entity;
173 		if (request.hasBody()) {
174 			InputStream stream = ((SenderBodyRequest) request).getBodyStream();
175 			if (stream instanceof FakeStream) {
176 				FakeStream fake = (FakeStream) stream;
177 				if (fake.getValue() instanceof String) {
178 					entity = new StringRequestEntity((String) fake.getValue(), null, charset.name());
179 				} else {
180 					RequestBodyMarshaller marshaller = getRequestMarshaller(mimeType);
181 					if (marshaller == null) {
182 						throw new IllegalArgumentException("Request body marshaller not found for " + mimeType);
183 					}
184 					entity = new ObjectEntity(fake.getValue(), charset, marshaller, fake.isStreaming());
185 				}
186 			} else {//plain InputStream
187 				entity = new InputStreamRequestEntity(stream);
188 			}
189 		} else if (query != null && query.length() != 0) {
190 			entity = new ByteArrayRequestEntity(EncodingUtil.getBytes(query, config.getEncoding()),
191 					PostMethod.FORM_URL_ENCODED_CONTENT_TYPE);
192 		} else {
193 			logger.debug("Body request does not have any parameters or body");
194 			entity = new StringRequestEntity("", null, config.getEncoding());
195 			//throw new IllegalArgumentException("POST request does not have any parameters or body");
196 		}
197 		return entity;
198 	}

200 	protected int call(HttpMethodBase httpRequest) throws IOException {
201 		try {
202 			return this.httpClient.executeMethod(httpRequest);
203 		} catch (Exception x) {
204 			//connection might be already open so release request
205 			httpRequest.releaseConnection();
206 			//now try to 
207 			if (x instanceof ConnectionPoolTimeoutException) {
208 				ConnectException cx = new ConnectException("Pool timeout " + config.getPoolAcquireTimeoutMillis() + " ms");
209 				cx.setStackTrace(x.getStackTrace());
210 				throw cx;
211 			} else if (x instanceof ConnectTimeoutException) {
212 				ConnectException cx = new ConnectException("Connect timeout " + config.getConnectTimeoutMillis() + " ms");
213 				cx.setStackTrace(x.getStackTrace());
214 				throw cx;
215 			} else if (x instanceof SocketTimeoutException) {
216 				int timeout = httpRequest.getParams().getIntParameter(HttpMethodParams.SO_TIMEOUT,
217 						config.getReadTimeoutMillis());
218 				SocketTimeoutException stx = new SocketTimeoutException("Read timeout " + timeout + " ms");
219 				stx.setStackTrace(x.getStackTrace());
220 				throw stx;
221 			} else if (x instanceof ConnectException) {
222 				//enhance message with url
223 				ConnectException ctx = new ConnectException("Connection refused " + config.getHostUrl());
224 				ctx.setStackTrace(x.getStackTrace());
225 				throw ctx;
226 			} else if (x instanceof IOException) {
227 				throw (IOException) x;//just rethrow IO
228 			} else {
229 				throw new IOException(x.getMessage(), x);//wrap others
230 			}
231 		}
232 	}

273 	private static class ObjectEntity implements RequestEntity {
274 
275 		private final Object objectBody;
276 
277 		private final Charset charset;
278 
279 		private byte[] content;
280 
281 		private final RequestBodyMarshaller marshaller;
282 
283 		private ObjectEntity(Object objectBody, Charset charset, RequestBodyMarshaller marshaller, boolean streaming)
284 				throws IOException {
285 			this.objectBody = objectBody;
286 			this.marshaller = marshaller;
287 			this.charset = charset;
288 			if (streaming) {
289 				this.content = null;
290 			} else {
291 				ByteArrayOutputStream baos = new ByteArrayOutputStream();
292 				marshaller.write(objectBody, baos, charset);
293 				this.content = baos.toByteArray();
294 			}
295 		}
296 
297 		@Override
298 		public boolean isRepeatable() {
299 			return true;
300 		}
301 
302 		@Override
303 		public void writeRequest(OutputStream outstream) throws IOException {
304 			if (this.content != null) {
305 				outstream.write(content, 0, content.length);
306 				outstream.flush();
307 			} else {
308 				//streaming
309 				marshaller.write(objectBody, outstream, charset);
310 			}
311 
312 		}
313 
314 		@Override
315 		public long getContentLength() {
316 			return this.content != null ? content.length : -1;
317 		}
318 
319 		@Override
320 		public String getContentType() {
321 			return null; //we set ContentType before
322 		}
