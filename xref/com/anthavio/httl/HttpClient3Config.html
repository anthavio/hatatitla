43  		clientParams.setConnectionManagerTimeout(poolAcquireTimeoutMillis); //http.connection-manager.timeout
44  
45  		MultiThreadedHttpConnectionManager connectionManager = new MultiThreadedHttpConnectionManager();
46  		HttpConnectionManagerParams managerParams = new HttpConnectionManagerParams();
47  		HostConfiguration hostConfig = new HostConfiguration();
48  		hostConfig.setHost(getHostUrl().getHost(), getHostUrl().getPort(), getHostUrl().getProtocol());
49  		managerParams.setMaxConnectionsPerHost(hostConfig, getPoolMaximumSize());
50  		managerParams.setMaxTotalConnections(getPoolMaximumSize());
51  		managerParams.setConnectionTimeout(getConnectTimeoutMillis());//http.connection.timeout
52  		connectionManager.setParams(managerParams);
53  
54  		HttpClient httpClient = new HttpClient(clientParams, connectionManager);
55  		httpClient.setHostConfiguration(hostConfig);
56  
57  		//http://hc.apache.org/httpclient-3.x/authentication.html
58  		if (getAuthentication() != null) {
59  			Authentication authentication = getAuthentication();
60  
61  			UsernamePasswordCredentials credentials = new UsernamePasswordCredentials(authentication.getUsername(),
62  					authentication.getPassword());
63  			AuthScope scope = new AuthScope(getHostUrl().getHost(), getHostUrl().getPort(), authentication.getRealm()/*, authentication
64  																																																								.getScheme().toString()*/);
65  			httpClient.getState().setCredentials(scope, credentials);
66  
67  			if (authentication.getPreemptive() && authentication.getScheme() == Scheme.BASIC) {
68  				//Only BASIC can be preemtive

70  			}
