153 		public void handle(String target, HttpServletRequest request, HttpServletResponse response, int dispatch)
154 				throws IOException, ServletException {
155 			requestCount.incrementAndGet();
156 			Date now = new Date();
157 
158 			//request.setCharacterEncoding("utf-8");
159 			String hsleep = request.getHeader("sleep");
160 			if (hsleep != null) {
161 				sleep(hsleep, request);
162 			}
163 			String psleep = request.getParameter("sleep");
164 			if (psleep != null) {
165 				sleep(psleep, request);
166 			}
167 
168 			if (httpCode != HttpURLConnection.HTTP_OK) {
169 				response.sendError(httpCode);
170 				((Request) request).setHandled(true);
171 				return;
172 			}
173 
174 			String docache = request.getParameter("docache");
175 			if (docache != null) {
176 				int seconds = Integer.parseInt(docache);
177 				Calendar calendar = GregorianCalendar.getInstance();
178 				calendar.setTime(now);
179 				calendar.add(Calendar.SECOND, seconds);
180 				//HTTP 1.0 - Expire header
181 				response.setDateHeader("Date", now.getTime());
182 				response.setDateHeader("Expire", calendar.getTimeInMillis());
183 				//HTTP 1.1 - Cache-Control header
184 				response.setHeader("Cache-Control", "private, max-age=" + seconds);
185 				writeResponse(request, response, now, "Docache for " + seconds + " seconds");
186 				return;
187 			}
188 
189 			String doetag = request.getParameter("doetag");
190 			if (doetag != null) {
191 				String cetag = request.getHeader("If-None-Match");
192 				if (etag.equals(cetag)) {
193 					response.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
194 					((Request) request).setHandled(true);
195 					return; //no response...
196 				} else {
197 					response.setHeader("ETag", etag);
198 					//we need unique content - put date into it
199 					String message = "Doetag  for " + etag + " ETag";
200 					writeResponse(request, response, now, message);
201 					return;
202 				}
203 			}
204 
205 			String pstatus = request.getParameter("dostatus");
206 			if (pstatus != null) {
207 				int status = Integer.parseInt(pstatus);
208 				response.setStatus(status);
209 				writeResponse(request, response, now, "Dostatus " + status);
210 				return;
211 			}
212 
213 			if (((Request) request).isHandled() == false) {
214 				response.setStatus(HttpServletResponse.SC_OK);
215 				writeResponse(request, response, now, "Hello");
216 			}
217 			//((Request) request).setHandled(true);
218 		}

337 	private SecurityHandler getBasicSecurityHandler(UserRealm realm, String role) {
338 
339 		Constraint constraint = new Constraint(Constraint.__BASIC_AUTH, role);
340 		constraint.setAuthenticate(true);
341 
342 		ConstraintMapping mapping = new ConstraintMapping();
343 		mapping.setConstraint(constraint);
344 		mapping.setPathSpec("/basic/*");
345 
346 		SecurityHandler handler = new SecurityHandler();
347 		handler.setAuthenticator(new BasicAuthenticator());
348 		handler.setUserRealm(realm);
349 		handler.setConstraintMappings(new ConstraintMapping[] { mapping });
350 
351 		return handler;
